44  void fromEuler(f32 pitch, f32 yaw, f32 roll) {
45  // Basically we create 3 Quaternions, one for pitch, one for yaw, one for roll
46  // and multiply those together.
47  // the calculation below does the same, just shorter
48 
49  const f32 PIOVER180 = M_PI / 180.0f;
50 
51  f32 p = pitch * PIOVER180 / 2.0f;
52  f32 y = yaw * PIOVER180 / 2.0f;
53  f32 r = roll * PIOVER180 / 2.0f;
54 
55  f32 sinp = sin(p);
56  f32 siny = sin(y);
57  f32 sinr = sin(r);
58  f32 cosp = cos(p);
59  f32 cosy = cos(y);
60  f32 cosr = cos(r);
61 
62  q.x = sinr * cosp * cosy - cosr * sinp * siny;
63  q.y = cosr * sinp * cosy + sinr * cosp * siny;
64  q.z = cosr * cosp * siny - sinr * sinp * cosy;
65  q.w = cosr * cosp * cosy + sinr * sinp * siny;
66 
67  normalize();
68  }
